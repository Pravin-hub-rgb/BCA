#include <stdio.h>
#include <math.h>
#include <stdlib.h> // so we can use exit(0) function
#define EPSILON 0.003

// Function which will give value of equation
float f(float x)
{
    return ((x * x * x) - (9 * x) + 1);
}

int main()
{
    float x1, x2, xmid;
    int i = 1, itr;
    printf("Enter Initial Guess: ");
    scanf("%f", &x1);
    scanf("%f", &x2);

    if (f(x1) * f(x2) > 0)
    {
        printf("\nAssumptions are wrong!");
        exit(0);
    }
    printf("\nEnter number of iteration : ");
    scanf("%d", &itr);
    do
    {
        xmid = (x1 + x2) / 2;
        printf("\nxmid is = %f", xmid);
        if (f(xmid) == 0.0)
        {
            printf("\nRoot is = %f", xmid);
            exit(0);
        }
        if (f(xmid) * f(x1) < 0)
            x2 = xmid;
        else
            x1 = xmid;
        printf("\nRoot after %d iteration = %f\n", i, xmid);
        i = i + 1;
    } while (fabs(x1 - x2) > EPSILON && i <= itr);

    return 0;
}
